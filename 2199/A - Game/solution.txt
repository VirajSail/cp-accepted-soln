import java.util.*
 
fun main() {
    val sc = Scanner(System.`in`)
    val t = sc.nextInt()
    repeat(t) {
        val k = sc.nextInt()
        val a = sc.nextInt()
        val b = sc.nextInt()
        val x = sc.nextInt()
        val y = sc.nextInt()
        if(a + x > k + b + y) {
            println("NO")
        }
        else if(a > b && x > y && a + x == b + k + y) {
            println("NO")
        }
        else {
            println("YES")
        }
    }
}
